\chapter{Конус устойчивости для линейного дифференциального уравнения с
запаздыванием}
\section{Устойчивость скалярного аналога}

Различные критерии устойчивости матричного уравнения
\begin{equation}\label{main}
    \dot{x}(t) + A\, x(t) +  B\, x(t-\tau) = 0,\quad \tau > 0.
\end{equation}
даны в
\cite{Gu&Khar_2003,Kiryanen_1994,Kiryanen&Gal_1989,Mori&Fukuma&Kuwahara_1981,Mori&Kokame_1989,Wang_1992}.
В \cite{Matsunaga_2009} исследуется эта задача для матриц $2\times 2$
специального вида. В этой заметке мы даем метод, позволяющую дать полный
геометрический обзор тех значений запаздывания $\tau$, который обеспечивают 
устойчивость уравнения \eqref{main}. Метод пригоден для уравнений с матрицами
$A,B$, которые могут быть приведены одной трансформирующей матрицей к
треугольному виду. Это в точности класс матриц $A,B$, таких что $AB-BA$
нильпотентна \cite{Horn&Johnson_1986}. Этот класс включает все коммутирующие
пары матриц $A,B$.

\section{Основной результат}

\textbf{Определение 1.} Конус устойчивости для уравнения \eqref{main} это 
множество точек $M=(u_1,u_2,u_3)$, таких что
\begin{equation}\label{cone}
    \left\{
        \begin{array}{l}
            u_1=-a \cos{\omega} + \omega \sin{\omega},\\
            u_2=a \sin{\omega} + \omega \cos{\omega},\\
            u_3 = a,
        \end{array}
    \right.
\end{equation}
где параметры $a, \omega$ пробегают следующую  область  в $\mathbb{R}^2$:
\begin{equation}\label{preimage}
    \left\{
        \begin{array}{l}
            a\geqslant -\dfrac{\omega}{\tan{\omega}},\\[2mm]
            -\pi<\omega<\pi .\\
        \end{array}
    \right.
\end{equation}

Единственный конус устойчивости \eqref{cone}, \eqref{preimage} строится для 
класса всех уравнений вида \eqref{main}.\\
\textbf{Теорема 1}
Пусть $A, B, S\in \mathbb{R}^{m\times m}$ и $S^{-1}AS=A_{T}$ и $S^{-1}BS=B_{T}$,
где $A_{T}$ и $B_{T}$ - нижние треугольные матрицы с элементами соответственно 
$\lambda_{j\,s}, \mu_{j\,s}\,(1 \leqslant  j,\, s \leqslant  m)$. Построим 
систему точек $M_j=(u_{1j},u_{2j},u_{3j})$, $(1\leqslant j \leqslant m)$,  так,
что
\begin{equation}\label{pointsM}
    \begin{array}{l}
        u_{1j} = \tau |\mu_{jj}| \cos (\arg \mu_{jj}+\tau |\lambda_{j j}| 
        \sin \arg \lambda_{j j}), \\
        u_{2j} = \tau |\mu_{jj}| \sin (\arg \mu_{jj}+\tau |\lambda_{j j}| 
        \sin \arg \lambda_{j j}), \\
        u_{3j} = \tau |\lambda_{jj}| \cos \arg \lambda_{jj}.
    \end{array}
\end{equation}
Уравнение \eqref{main} асимптотически устойчиво тогда и только тогда,  когда все
точки $M_j$ $(1\leqslant j \leqslant m)$ находятся внутри конуса устойчивости 
(см. Определение 1). Если хотя бы одна точка $M_j$  $(1\leqslant j \leqslant m)$
лежит вне конуса устойчивости, то уравнение \eqref{main} неустойчиво.

\medskip

\textbf{Замечание 1.} Взамен \eqref{pointsM} точки $M_j$, можно определить 
равенствами
\begin{equation}\label{otherpointsM}
    \begin{array}{l}
        u_{1j} = \tau \Re (\mu_{jj}\exp(i \tau \Im \lambda_{j j})), \\
        u_{2j} = \tau \Im (\mu_{jj}\exp(i \tau \Im \lambda_{j j})), \\
        u_{3j} = \tau \Re \lambda_{jj}.
    \end{array}
\end{equation}

\section{Примеры}%(Пример Михаила Марковича)

\textbf{Пример 1.} С помощью геометрического метода Теоремы 1 найдем значения 
запаздывания $\tau$, обеспечивающие асимптотическую устойчивость \eqref{main} с
матрицами

 
\begin{equation}
    \label{matr5}
    A=
    \begin{pmatrix}
        0.72&-0.96\\
        4.08&-0.60
    \end{pmatrix},  \qquad
    B= 
    \begin{pmatrix}
        0.08&-0.08\\
        0.34&-0.03
    \end{pmatrix}.
\end{equation}
Поскольку $A=12B-0.24I$, матрицы $A,B$ коммутируют и, следовательно, могут быть
совместно приведены к диагональной форме \cite{HD}. Собственные числа матриц
$A,B$ суть соответственно
\begin{equation}
\label{eigenvalues4}
\lambda_{11, 22}=    0.0600 \pm 1.8658i,\quad \mu_{11, 22}=0.0250 \pm 0.1555i.
\end{equation}
Вследствие симметрии изучим расположение конической винтовой линии
\eqref{pointsM} относительно конуса устойчивости \eqref{cone},\eqref{preimage}
только при $j=1$. Кривая имеет вид (см. Рис. \ref{fig1}, \ref{fig1a})
\begin{equation}\label{pointsMex1}
    \begin{array}{l}
        u_{11} = 0.1575\tau \cos( 1.4114+ 1.8658\tau), \\
        u_{21} = 0.1575\tau \sin( 1.4114+ 1.8658\tau), \\
        u_{31} =  0.06 \tau .
    \end{array}
\end{equation}

Точки кривой находятся внутри конуса устойчивости при следующих значениях

9.64)\cup(12.66, 12.75)$, поэтому уравнение \eqref{main}  с матрицами 
\eqref{matr5} асимптотически устойчиво только при таких значениях запаздывания.

\textbf{Пример 2.} Положим в \eqref{main}
\begin{equation}
    A =
    \begin{pmatrix}
          1  &   1   &   -3  &  2\\
         -3 &   1   &   -2  &  3\\
         -1&   0.5   &   4  &  -2   \\
         -3 &   2   &   -1  &   1   \\
    \end{pmatrix},
\end{equation}
\begin{equation}
    B = 0.8I + 0.2A + 2A^{2}.
\end{equation}

Требуется определить, при каких значениях $\tau$ это уравнение устойчиво, а при
каких неустойчиво.
\bigskip

Очевидно, матрицы $A, B$ коммутируют. Собственные числа матриц $A$ и $B$ имеют
вид
\begin{equation}
    \begin{matrix}
        \lambda_{1}=3.5620,        &            \mu_{1}=26.8877,     \\
        \lambda_{2,3}=2.3677 \pm 1.5138i,    &   \mu_{2,3}= 7.9025 \pm 14.6403i,\\
        \lambda_{4}= -1.2975,                &   \mu_{4}= 3.9073 .
    \end{matrix}
\end{equation}

При этом
\begin{equation}
    \begin{array}{l}
        a_{1}=|\lambda_{1}|\cos{\arg{\lambda_{1}}}=\Re{\lambda_{1}}=3.562, \\
        a_{2}=|\lambda_{2}|\cos{\arg{\lambda_{2}}}=\Re{\lambda_{2}}=2.3677, \\
        a_{3}=|\lambda_{3}|\cos{\arg{\lambda_{3}}}=\Re{\lambda_{3}}=2.3677, \\
        a_{4}=|\lambda_{4}|\cos{\arg{\lambda_{4}}}=\Re{\lambda_{4}}=-1.2975.
    \end{array}
\end{equation}

\begin{equation}
    \begin{array}{c}
        b_{1} = |\mu_{1}|\,\exp(i(\arg \mu_{1}+|\lambda_{1}|\tau \sin{\arg 
        \lambda_{1}}))=26.8877,\\
        
        \smallskip
        b_{2} = |\mu_{2}|\,\exp(i(\arg \mu_{2}+|\lambda_{2}|\tau \sin{\arg 
        \lambda_{2}}))=\\ =16.6369 \exp(i(1.0758+ 1.5138\tau)),\\
        
        \smallskip
        b_{3} = |\mu_{3}|\,\exp(i(\arg \mu_{3}+|\lambda_{3}|\tau \sin{\arg 
        \lambda_{3}}))=\\ =16.6369 \exp(i(- 1.0758- 1.5138\tau)),\\
        
        \smallskip
        b_{4} = |\mu_{4}|\,\exp(i(\arg \mu_{4}+|\lambda_{4}|\tau \sin{\arg 
        \lambda_{4}}))= 3.9073.
    \end{array}
\end{equation}

Построим точки $(\Re b_{j}\tau,\, \Im b_{j}\tau,\, a_{j}\tau), j=\overline{1,
4}$ и конус устойчивости. Они изображены на Рис.\,2.

Для устойчивости необходимо и достаточно, чтобы все точки $(\Re b_{j}\tau,\, 
\Im b_{j}\tau,\, a_{j}\tau)$ находились внутри конуса устойчивости. Две системы
точек, соответствующие комплексным собственным значениям матрицы $A$, с
изменением $\tau$ образуют винтовые линии, пересекающие конус устойчивости при
$\tau_{2,3}\approx0.033$. Две другие системы точек, соответствующие 
действительным собственным значениям матрицы $A$, находятся внутри конуса при
$\tau\in(0,\tau_{1})$, где $\tau_{1}\approx0.063$, и при $\tau\in(0,\tau_{4})$,
где $\tau_{4}\approx0.321$.

Итак, уравнение \eqref{15.1} устойчиво тогда и только тогда, когда
$\tau\in(0,\tau_{0})$, где $\tau_{0}=\min(\tau_{1},\tau_{2,3},
\tau_{4})=\tau_{2,3}\approx0.033$.
